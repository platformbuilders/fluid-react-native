# KeyboardAvoidingView

## Usage

```jsx
import React from 'react';
import { KeyboardAvoidingView, SceneWrapper } from '@platformbuilders/fluid-react-native';

const Home: React.FC = () => {
   <KeyboardAvoidingView>
      <SceneWrapper>{...}</SceneWrapper>
   </KeyboardAvoidingView>
  );
}

export default Home;
```
