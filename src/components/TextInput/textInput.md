## Source Code
```js
import React from 'react';
import { TextInput } from '@platformbuilders/react-native-ui';

const Home: React.FC = () => {
  return (
      <TextInput
        large
        id="products_searc"
        accessibility="Product Search"
        placeholder="Product Search "
        contrast
        centered
        multiline
        iconTouchableEnabled
        iconName="barcode"
        iconSize={40}
        label="Products"
        error="has a error"
      />
  );
}

export default Home;
```
